\begin{tikzpicture}

  \draw[domain1] (p0) -- (p1) -- (p2) -- (p3) -- (p0);

  \node[annotation, below right] at (p1) {$(x_1,y_0)$};

  \foreach \x in {0,...,2}{%

    }

  \foreach \y in {0,...,1}{%

    }
